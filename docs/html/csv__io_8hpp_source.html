101 /** \brief function to escape characters in csv fields according to RFC 4180
102  *
103  * Source: https://tools.ietf.org/html/rfc4180
104  *
105  * If discovered that the string doesn't have any characters which need to be escaped,
106  * the string will not be enclosed in quotes
107  *
108  * Also handles alternative delimiters and allows forced escapes
109  * which will append leading and trailing double quotes
110  *
111  * Generally O(n) avg time and space complexity
112  *
113  * \param data string to escape
114  * \param delim used to check if a delimiter is in the field. if so, it should be escaped
115  * \param force_escape wrap the string in quotes even if not needed
116  * \return escaped csv field if necessary, otherwise the string
117  */
118 inline string escape(string_view data, char delim = ',', bool force_escape = false) {
119  string result;
120  result.reserve(
121  data.length()); // avoid lots of allocations by setting a reasonable initial allocation
122 
123  bool needs_escape{false};
124  for (const char& c : data) {
125  switch (c) {
126  case '\"':
127  result.push_back(c);
128  case '\r':
129  case '\n':
130  needs_escape = true;
131  break;
132  default:
133  if (c == delim) needs_escape = true; // special handle for alternative delimiters
134  }
135  result.push_back(c);
136  }
137 
138  if (needs_escape || force_escape) {
139  result.push_back('\"');
140  result.insert(0, "\"");
141  }
142 
143  return result;
144 }

146 /** \brief unescape a csv escaped string according to RFC 4180
147  *
148  * Generally linear avg time and space complexity
149  *
150  * \param data string_view csv data to unescape
151  * \return unquoted csv field
152  */
153 inline string unescape(string_view data) {
154  string result;
155  int quotes_seen = 0;
156 
157 #if defined(__cpp_lib_string_view) || defined(__cpp_lib_experimental_string_view)
158  if (data[0] == '\"') { // if the first char is a quote, then the string is assumed to be quoted
159  data = data.substr(1, data.size() - 2);
160  }
161  string_view sv_data = data;
162 #else
163  // keep const string& (which is "string_view" now) construction, and if statement.
164  string_view sv_data = (data[0] == '\"') ? data.substr(1, data.size() - 2) : data;
165 #endif
166 
167  for (const char& c : sv_data) {
168  switch (c) {
169  case '\"':
170  quotes_seen++;
171  if (quotes_seen == 2) {
172  quotes_seen = 0;
173  result.push_back('\"');
174  }
175  break;
176  default:
177  quotes_seen = 0;
178  result.push_back(c);
179  }
180  }
181  return result;
182 }

191 struct DelimSplitNaive {
192  /** \brief Split string_view csv row based on a delimiter
193  *
194  * Linear time complexity with respect to the size of input
195  *
196  * \param input string_view to split by delimiter
197  * \param delim delimiter to split on
198  * \return RowContainer of split csv fields
199  */
200  RowContainer<string>& operator() (string_view input, const char delim) {
201  static_assert(
202  has_push_back<
203  RowContainer<string>, void(const typename RowContainer<string>::value_type&)>::value,
204  "The template parameter needs to be a container type with an push_back function.");
205  m_data.clear();
206 
207  string tmp(input.data(), input.length());
208 
209  size_t first = 0;
210  while (first < tmp.size()) {
211  const auto second = tmp.find_first_of(delim, first);
212  if (first != second) m_data.push_back(tmp.substr(first, second - first));
213  if (second == string::npos) break;
214  first = second + 1;
215  }
216  if (m_data.empty()) m_data.push_back("");
217  return m_data;
218  }
