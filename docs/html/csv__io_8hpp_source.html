84 /** \brief function to escape characters in csv fields according to RFC 4180
85  *
86  * Source: https://tools.ietf.org/html/rfc4180
87  *
88  * If discoverd that the string doesn't have any characters which need to be escaped,
89  * the string will not be enclosed in quotes
90  *
91  * Also handles alternative delimiters and allows forced escapes
92  * which will append leading and trailing double quotes
93  *
94  * Generally O(n) avg time and space complexity
95  *
96  * \param data string to escape
97  * \param delim used to check if a delimiter is in the field. if so, it should be escaped
98  * \param force_escape wrap the string in quotes even if not needed
99  * \return escaped csv field if necessary, otherwise the string
100  */
101 inline std::string escape(std::string_view data, char delim = ',', bool force_escape = false) {
102  std::string result;
103  result.reserve(
104  data.length()); // avoid lots of allocations by setting a reasonable initial allocation
105 
106  bool needs_escape{false};
107  for (const char& c : data) {
108  switch (c) {
109  case '\"':
110  result.push_back(c);
111  case '\r':
112  case '\n':
113  needs_escape = true;
114  break;
115  default:
116  if (c == delim) needs_escape = true; // special handle for alternative delimiters
117  }
118  result.push_back(c);
119  }
120 
121  if (needs_escape || force_escape) {
122  result.push_back('\"');
123  result.insert(0, "\"");
124  }
125 
126  return result;
127 }

129 /** \brief unescape a csv escaped string according to RFC 4180
130  *
131  * Generally linear avg time and space complexity
132  *
133  * \param data string_view csv data to unescape
134  * \return unquoted csv field
135  */
136 inline std::string unescape(std::string_view data) {
137  std::string result;
138  int quotes_seen = 0;
139 
140  if (data[0] == '\"') { // if the first char is a quote, then the string is assumed to be quoted
141  data = data.substr(1, data.size() - 2);
142  }
143 
144  for (const char& c : data) {
145  switch (c) {
146  case '\"':
147  quotes_seen++;
148  if (quotes_seen == 2) {
149  quotes_seen = 0;
150  result.push_back('\"');
151  }
152  break;
153  default:
154  quotes_seen = 0;
155  result.push_back(c);
156  }
157  }
158  return result;
159 }

164  * \brief Class of functions to parse a csv row into a non map container

168 template <template <class...> class RowContainer>
169 struct CSVInputParser {
170  static RowContainer<std::string> m_data;
171  /** \brief Split string_view csv row based on a delimiter
172  *
173  * Linear time complexity with respect to the size of input
174  *
175  * \param input string_view to split by delimiter
176  * \param delim delimiter to split on
177  * \return RowContainer of split csv fields
178  */
179  static RowContainer<std::string> delim_split_naive(std::string_view input, const char delim) {
180  static_assert(
181  has_push_back<
182  RowContainer<std::string>,
183  void(const typename RowContainer<std::string>::value_type&)>::value,
184  "The template parameter needs to be a container type with an push_back function.");
185  m_data.clear();
186 
187  std::string tmp(input.data(), input.length());
188 
189  size_t first = 0;
190  while (first < tmp.size()) {
191  const auto second = tmp.find_first_of(delim, first);
192  if (first != second) m_data.push_back(tmp.substr(first, second - first));
193  if (second == std::string_view::npos) break;
194  first = second + 1;
195  }
196  if (m_data.empty()) m_data.push_back("");
197  return m_data;
198  }

200  /** \brief implementation of delim_split_escaped
201  *
202  * Linear time complexity with respect to the size of input
203  *
204  * \param input string_view to split by delimiter
205  * \param delim delimiter to split on
206  * \return RowContainer of split csv fields, if input string empty, return RowContainer with one
207  * empty string
208  */
209  static void delim_split_escaped_impl(std::string_view input, const char delim) {
210  static_assert(
211  has_push_back<
212  RowContainer<std::string>,
213  void(const typename RowContainer<std::string>::value_type&)>::value,
214  "The template parameter needs to be a container type with an push_back function.");
215 
216  CSVParserScope state{LINE};
217 
218  std::string chunk;
219  chunk.reserve(512);
220  int num_cols{1};
221 
222  for (const auto& c : input) {
223  if (state == LINE) {
224  if (c == '\"') {
225  state = QUOTE;
226  chunk.push_back(c);
227  } else {
228  if (c == delim || c == '\n') {
229  if (c == delim) num_cols++;
230  m_data.push_back(chunk);
231  chunk.clear();
232  } else {
233  chunk.push_back(c);
234  }
235  }
236  } else {
237  if (c == '\"') state = LINE;
238  chunk.push_back(c);
239  }
240  }
241  if (!chunk.empty() || m_data.size() < num_cols) m_data.push_back(chunk);
242 
243  if (auto& last_element = m_data.back(); last_element.back() == '\r') last_element.pop_back();
244  }

288  * \param delim delimiter to split on
289  * \return RowContainer of split csv fields, if input string empty, return RowContainer with one

393  * \brief Stateful CSV line reader which reads csv lines with escaped fields according to RFC 4180
394  */
395 class CSVLineReader {
396 public:
397  /** \brief Construct a CSVLineReader from a reference to a std::istream
398  * \param instream reference to a std::istream
399  */
400  explicit CSVLineReader(std::istream& instream) : m_csv_stream(instream) {}
401 
402  /** \brief read a csv line
403  * \return a string with the contents of the csv line
404  */
405  std::string readline() {
406  m_result.clear();
407 
408  char buf[1024]{0};
409  char* buf_end = buf + 1023;
410  char* pos = buf;
411 
412  while (m_csv_stream.good()) {
413  m_csv_stream.get(*pos);
414  if (m_state == LINE && (pos[0] == '\n' || pos[0] == EOF)) {
415  *(++pos) = '\0';
416  m_result.append(buf);
417  pos = buf;
418 // std::fill(buf, buf + 1024, '\0');
419  break;
420  } else if (m_state == LINE && pos[0] == '\"') {
421  m_state = QUOTE;
422  } else if (m_state == QUOTE && pos[0] == '\"') {
423  m_state = LINE;
424  } else if (m_state == QUOTE && !m_csv_stream.good()) {
425  m_result = "";
426  return m_result;
427  }
428  pos++;
429 
430  if (pos >= buf_end) {
431  *(pos++) = '\0';
432  m_result.append(buf);
433  pos = buf;
434  //std::fill(buf, buf + 1024, '\0');
435  }
436  }
437  if (pos != buf) {
438  *(pos++) = '\0';
439  m_result.append(buf);
440  }
441 
442  m_lines_read++;
443  m_state = LINE;
444  return m_result;
445  }
446 
447  /** \brief Get number of csv lines read so far
448  * \return number of csv lines read so far
449  */
450  const size_t lcount() const { return m_lines_read; }
451 
452  /** \brief check if the underlying stream is still good
453  * \return true if good, otherwise false
454  */
455  bool good() { return m_csv_stream.good(); }
456 
457 private:
458  CSVParserScope m_state{LINE};
459  std::istream& m_csv_stream;
460  std::string m_result;
461  size_t m_lines_read{0};
